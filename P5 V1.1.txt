declare -i pick=-1

main () {
        clear
        whileLoop
        exitMenu
}

whileLoop () {
        while [ "$pick" -ne 0 ]
        do
                printMenu
                setPick
                checkPick
                main
        done
}

setPick () {
                echo ""
                echo "Write your selection then press enter."
                echo ""
                read -p "Your Selection: " pick
}

printMenu () {
        echo "1.) Editing Commands"
        echo "2.) Games"
        echo "3.) User Administration"
        echo "4.) Process Commands"
        echo "5.) System Commands"
        echo "6.) Network Commands"
        echo ""
        echo "Press 0 to exit."
}

checkPick () {
        if [ $pick == 1 ]
        then
                editCommandMenu
        elif [ $pick == 2 ]
        then
                gameMenu
        elif [ $pick == 3 ]
        then
                userAdminMenu
        elif [ $pick == 4 ]
        then
                processCommandMenu
        elif [ $pick == 5 ]
        then
                systemCommandMenu
        elif [ $pick == 6 ]
        then
                networkCommandMenu
        elif [ $pick == 0 ]
        then
                main
        else
                echo "Invalid option. Please select a valid option."
                sleep 2
        fi
}

editCommandMenu () {
        clear
        echo "Command Editing Menu"
        echo ""
        echo "Please wait for a future version"
        sleep 5
}

gameMenu () {
        clear
        echo "Game Menu"
        echo ""
        echo "1. Pacman"
        echo "2. Minesweeper"
        echo "3. Zerg Rush"
        echo ""
        echo "Press 9 to return to the main menu."
        setPick
        checkGame
}

checkGame () {
        if [ $pick = 1 ]
        then
                pacmanGame
        elif [ $pick = 2 ]
        then
                minesweeperGame
        elif [ $pick = 3 ]
        then
                zergRushGame
        elif [ $pick = 0 ]
        then
                pick=-1
                checkGame
        elif [ $pick = 9 ]
        then
                main
        else
                clear
                echo "Invalid option!"
                sleep 2
                gameMenu
        fi
}

        clear
        echo "Pacman"
        echo ""
